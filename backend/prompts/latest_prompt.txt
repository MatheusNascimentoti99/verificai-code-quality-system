export { IS_PUBLIC_KEY, Public };
import { Request } from 'express';
import { CanActivate, ExecutionContext, Injectable, UnauthorizedException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { JwtService } from '@nestjs/jwt';
import { IS_PUBLIC_KEY } from './public';
import { AppLogger } from '../logger';
import { AuthSessionStore } from '@/utils/auth-session.store';
import { getDataSource } from '@/infrastructure/databases/typeorm/postgres/database.providers';

declare module 'express-serve-static-core' {
  interface Request {
    payload?: JwtPayload;
    session?: any;
    user?: any;
  }
}

interface JwtPayload {
  exp?: number;
  sub?: string;
  [key: string]: unknown;
}

@Injectable()
export class AuthGuard implements CanActivate {
  constructor(
    private readonly jwtService: JwtService,
    private readonly reflector: Reflector,
    private readonly logger: AppLogger
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const isPublic = this.reflector.getAllAndOverride<boolean>(IS_PUBLIC_KEY, [
      context.getHandler(),
      context.getClass()
    ]);

    const request = context.switchToHttp().getRequest<Request>();
    const path = request.url;

    this.logger.log(`AuthGuard - Path: ${path}, isPublic: ${isPublic}`);

    if (isPublic) {
      this.logger.log(`Rota pública detectada - permitindo acesso: ${path}`);
      return true;
    }

    try {
      const token = this.extractTokenFromHeader(request);
      const sessionId = this.extractSessionId(request);

      if (!token || !sessionId) {
        this.logger.warn(`Token ou sessionId ausente - Path: ${path}`);
        throw new UnauthorizedException();
      }

      const payload = this.validateToken(token);
      const session = await this.validateSession(sessionId, payload);

      request['payload'] = payload;
      request['session'] = session;

      if (session?.usuario) {
        request['user'] = session.usuario;
      }
    } catch (error) {
      if (error instanceof Error) {
        this.logger.error(error.message, error.stack, 'AuthGuard');
      } else {
        this.logger.error('Erro desconhecido ocorrido no AuthGuard', '', 'AuthGuard');
      }

      throw new UnauthorizedException();
    }

    return true;
  }

  private validateToken(token: string): JwtPayload {
    if (token === 'dev-access-token') {
      return {
        sub: '12312312300',
        exp: Math.floor(Date.now() / 1000) + 3600 // 1 hour from now
      };
    }

    try {
      const payload = this.jwtService.decode(token);

      if (!payload || typeof payload === 'string') {
        throw new UnauthorizedException('Token inválido');
      }

      const typedPayload = payload as JwtPayload;
      const expiration = typeof typedPayload.exp === 'number' ? typedPayload.exp * 1000 : undefined;

      if (expiration && Date.now() >= expiration) {
        throw new UnauthorizedException('Token expirado');
      }

      return typedPayload;
    } catch (error) {
      this.logger.error(
        'Validação do token falhou:',
        error instanceof Error ? error.message : String(error)
      );
      throw new UnauthorizedException('Token inválido');
    }
  }

  private async validateSession(sessionId: string, payload: JwtPayload) {
    if (sessionId.startsWith('sess_') && payload.sub === '12312312300') {
      this.logger.log('Usando sessão de desenvolvimento');

      const payloadCpf = payload.sub.replace(/\D/g, '');
      const dataSource = getDataSource();
      const userRepo = dataSource.getRepository(Usuario);

      let usuario = await userRepo.findOne({ where: { nrCpf: payloadCpf } });

      if (!usuario) {
        this.logger.log('Criando usuário de desenvolvimento');
        usuario = userRepo.create({
          nrCpf: payloadCpf,
          nmUsuario: 'Dev User'
        });
        await userRepo.save(usuario);
      }

      return {
        id: sessionId,
        usuario: usuario,
        dhExpiracaoSessao: new Date(Date.now() + 24 * 60 * 60 * 1000)
      };
    }

    let session = await AuthSessionStore.get(sessionId);

    if (!session) {
      this.logger.error('Sessão não encontrada no banco de dados:');

      const payloadCpf =
        typeof payload.sub === 'string' ? payload.sub.replace(/\D/g, '') : undefined;
      if (payloadCpf) {
        try {
          const dataSource = getDataSource();
          const userRepo = dataSource.getRepository(Usuario);
          const usuario = await userRepo.findOne({ where: { nrCpf: payloadCpf } });

          if (usuario) {
            const expiration =
              typeof payload.exp === 'number'
                ? payload.exp * 1000
                : Date.now() + 24 * 60 * 60 * 1000;
            await AuthSessionStore.set(sessionId, usuario.id, expiration);
            session = await AuthSessionStore.get(sessionId);
            this.logger.log('Sessão temporária criada com sucesso');
          } else {
            this.logger.error('Usuário não encontrado para CPF:', payloadCpf);
          }
        } catch (error) {
          this.logger.error(
            'Falha ao criar sessão temporária:',
            error instanceof Error ? error.message : String(error)
          );
        }
      }

      if (!session) {
        throw new UnauthorizedException('Sessão inválida');
      }
    }

    const sessionExpiration = session.dhExpiracaoSessao?.getTime();

    if (sessionExpiration && Date.now() >= sessionExpiration) {
      this.logger.error('Sessão expirada:', sessionId);
      await AuthSessionStore.delete(sessionId);
      throw new UnauthorizedException('Sessão expirada');
    }

    const payloadCpf = typeof payload.sub === 'string' ? payload.sub.replace(/\D/g, '') : undefined;
    const sessionCpf = session.usuario?.nrCpf?.replace(/\D/g, '') ?? undefined;

    if (payloadCpf && sessionCpf && payloadCpf !== sessionCpf) {
      this.logger.error('CPF mismatch - Token não pertence à sessão informada');
      throw new UnauthorizedException('Token não pertence à sessão informada');
    }

    return session;
  }

  private extractTokenFromHeader(request: Request): string | undefined {
    const [type, token] = request.headers.authorization?.split(' ') ?? [];
    return type === 'Bearer' ? token : undefined;
  }

  private extractSessionId(request: Request): string | undefined {
    const headerValue = request.headers['x-session-id'];

    if (typeof headerValue === 'string' && headerValue.trim()) {
      return headerValue.trim();
    }

    if (Array.isArray(headerValue) && headerValue.length > 0) {
      const first = headerValue[0]?.trim();
      if (first) {
        return first;
      }
    }

    const cookieHeader = request.headers.cookie;

    if (!cookieHeader) {
      return undefined;
    }

    const cookies = cookieHeader.split(';').map(cookie => cookie.trim());

    for (const cookie of cookies) {
      const [rawName, ...rest] = cookie.split('=');

      if (!rawName || rawName.toLowerCase() !== 'session_id') {
        continue;
      }

      const value = rest.join('=');

      if (value) {
        return decodeURIComponent(value);
      }
    }

    return undefined;
  }
}

interface ApproveUserRequest {
  userId: string;
  aprovadorId: string;
}

@Injectable()
export class ApproveUserService {
  constructor(
    @Inject(USUARIO_REPOSITORY)
    private usuarioRepository: Repository<Usuario>
  ) {}

  async execute({ userId, aprovadorId }: ApproveUserRequest): Promise<Usuario> {
    const usuario = await this.usuarioRepository.findOne({ where: { id: userId } });

    if (!usuario) {
      throw new NotFoundException('Usuário não encontrado');
    }

    if (usuario.stAprovado) {
      throw new BadRequestException('Usuário já está aprovado');
    }

    const aprovador = await this.usuarioRepository.findOne({ where: { id: aprovadorId } });

    if (!aprovador) {
      throw new NotFoundException('Aprovador não encontrado');
    }

    usuario.stAprovado = true;
    usuario.usuarioAprovador = aprovador;
    usuario.dhAprovacao = new Date();

    await this.usuarioRepository.save(usuario);

    return usuario;
  }
}

interface CreateExtractJobInput {
  lpco: string;
  laudoFile: Express.Multer.File;
  usuario: Usuario;
  forceReprocess?: boolean;
}

interface CreateExtractJobOutput {
  id: number;
  status: LaudoStatus;
  cached?: boolean;
  jsDadoBruto?: any;
  obLaudoUsuario?: string;
  nrLpco?: string;
}

@Injectable()
export class CreateExtractJobService {
  constructor(
    @Inject(LAUDO_REPOSITORY)
    private laudoRepository: Repository<LaudoOlivia>
  ) {}

  async execute({
    lpco,
    laudoFile,
    usuario,
    forceReprocess = false
  }: CreateExtractJobInput): Promise<CreateExtractJobOutput> {
    if (!laudoFile || !laudoFile.buffer) {
      throw new BadRequestException('Arquivo PDF inválido');
    }

    let laudoExistente = await this.laudoRepository.findOne({
      where: { nrLpco: lpco, usuario: { id: usuario.id } }
    });

    // Se forceReprocess for false e já está com dados extraídos (CONCLUIDO), retornar do cache
    if (
      !forceReprocess &&
      laudoExistente?.jsDadoBruto &&
      laudoExistente.status === LaudoStatus.CONCLUIDO
    ) {
      return {
        id: laudoExistente.id,
        status: LaudoStatus.CONCLUIDO,
        cached: true,
        jsDadoBruto: laudoExistente.jsDadoBruto,
        obLaudoUsuario: laudoExistente.obLaudoUsuario || undefined,
        nrLpco: laudoExistente.nrLpco
      };
    }

    // Se existe, atualizar; se não existe, criar novo
    let laudoSalvo: LaudoOlivia;
    if (laudoExistente) {
      laudoExistente.status = LaudoStatus.PENDENTE;
      laudoExistente.jsDadoBruto = null;
      laudoExistente.jsDadoUsuario = null;
      laudoExistente.erroIa = undefined;
      laudoSalvo = await this.laudoRepository.save(laudoExistente);
    } else {
      const novoLaudo = this.laudoRepository.create({
        nrLpco: lpco,
        jsDadoBruto: null,
        jsDadoUsuario: null,
        status: LaudoStatus.PENDENTE,
        usuario
      });
      laudoSalvo = await this.laudoRepository.save(novoLaudo);
    }

    return {
      id: laudoSalvo.id,
      status: LaudoStatus.PENDENTE,
      cached: false
    };
  }
}

@Injectable()
export class DataProcessingService {
  private readonly logger = new Logger(DataProcessingService.name);

  constructor(
    @Inject(LAUDO_REPOSITORY)
    private laudoRepository: Repository<LaudoOlivia>
  ) {}
  async execute({ lpco, chem_data }: DataProcessingDto) {
    const laudo = await this.laudoRepository.findOne({ where: { nrLpco: lpco } });

    if (!laudo) {
      throw new NotFoundException('Laudo não encontrado para o LPCO informado.');
    }

    const body = {
      lpco,
      chem_data: chem_data.quimico,
      sec_data: chem_data.cabecalho
    };

    const aiFileName = generateFileName(lpco, 'ia');

    try {
      const apiResponse = await iaApi.post('/data-processing', body, {
        responseType: 'stream',
        headers: { 'Content-Type': 'application/json' },
        maxBodyLength: Infinity,
        maxContentLength: Infinity
      });

      try {
        const fileBuffer = await new Promise<Buffer>((resolve, reject) => {
          const chunks: Buffer[] = [];
          apiResponse.data.on('data', (chunk: Buffer) => chunks.push(chunk));
          apiResponse.data.on('end', () => resolve(Buffer.concat(chunks)));
          apiResponse.data.on('error', reject);
        });

        await minioClient.putObject(MINIO_BUCKET, aiFileName, fileBuffer, fileBuffer.length, {
          'Content-Type': apiResponse.headers['content-type'] || 'application/pdf'
        });
      } catch (e) {
        this.logger.error('Erro ao salvar no MinIO:', e);
        throw new BadRequestException('Erro ao salvar arquivo no MinIO.');
      }

      const xMensagem = apiResponse.headers['x-mensagem'];
      const xLpco = apiResponse.headers['x-lpco'];

      laudo.obLaudoIa = aiFileName;
      laudo.jsDadoUsuario = chem_data;
      await this.laudoRepository.save(laudo);

      return {
        message: xMensagem || 'Arquivo recebido e salvo com sucesso!',
        lpco: xLpco,
        laudo: aiFileName
      };
    } catch (error: any) {
      throw new BadRequestException(
        error.response?.data || error.message || 'Erro ao processar a requisição'
      );
    }
  }
}

interface ExtractCallbackInput {
  laudoId: number;
  status: 'CONCLUIDO' | 'ERRO';
  mensagem?: string;
  dados_extraidos?: any;
  pdf_base64?: string;
}

interface ExtractCallbackOutput {
  message: string;
  laudo_id: number;
  status: LaudoStatus;
}

@Injectable()
export class ExtractCallbackService {
  constructor(
    @Inject(LAUDO_REPOSITORY)
    private laudoRepository: Repository<LaudoOlivia>
  ) {}

  async execute(input: ExtractCallbackInput): Promise<ExtractCallbackOutput> {
    const laudo = await this.laudoRepository.findOne({ where: { id: input.laudoId } });

    if (!laudo) {
      throw new NotFoundException('Laudo não encontrado');
    }

    try {

      if (input.status === 'CONCLUIDO') {
        laudo.status = LaudoStatus.CONCLUIDO;

        // Se recebeu PDF em base64, decodificar e salvar no MinIO
        if (input.pdf_base64) {
          try {
            const fileBuffer = Buffer.from(input.pdf_base64, 'base64');
            const fileName = generateFileName(laudo.nrLpco, 'ia');
            await minioClient.putObject(MINIO_BUCKET, fileName, fileBuffer, fileBuffer.length, {
              'Content-Type': 'application/pdf'
            });
            laudo.obLaudoIa = fileName;
          } catch (e) {
            throw new BadRequestException('Erro ao salvar arquivo no MinIO');
          }
        }

        // Salvar dados extraídos em jsDadoBruto
        if (input.dados_extraidos) {
          laudo.jsDadoBruto = input.dados_extraidos;
        }

        // Limpar erro anterior se houver
        laudo.erroIa = undefined;
      } else if (input.status === 'ERRO') {
        laudo.status = LaudoStatus.ERRO;
        laudo.erroIa = input.mensagem || 'Erro desconhecido na extração de dados';
      }

      await this.laudoRepository.save(laudo);

      return {
        message: 'Callback de extração processado com sucesso',
        laudo_id: laudo.id,
        status: laudo.status
      };
    } catch (error: any) {
      laudo.status = LaudoStatus.ERRO;
      laudo.erroIa = error.message || 'Erro ao processar callback de extração';
      await this.laudoRepository.save(laudo);

      throw new BadRequestException('Erro ao processar callback: ' + error.message);
    }
  }
}

interface ExtractPdfDataInput {
  laudoFile: Express.Multer.File;
  lpco: string;
  sub: string;
  forceReprocess?: boolean;
}

@Injectable()
export class ExtractPdfDataService {
  constructor(
    @Inject(LAUDO_REPOSITORY) private laudoRepository: Repository<LaudoOlivia>,
    @Inject(USUARIO_REPOSITORY) private usuarioRepository: Repository<Usuario>,
    private readonly logger: AppLogger
  ) {}
  async execute({
    laudoFile,
    lpco,
    sub,
    forceReprocess = false
  }: ExtractPdfDataInput): Promise<ExtractPdfDataDto> {
    if (!laudoFile) {
      throw new BadRequestException("Arquivo 'laudo' não foi enviado.");
    }

    if (!laudoFile.buffer || !laudoFile.size) {
      throw new BadRequestException('Arquivo inválido.');
    }

    const cpf = extractCpf(sub);

    let usuario = await this.usuarioRepository.findOne({ where: { nrCpf: cpf } });
    if (!usuario) {
      usuario = this.usuarioRepository.create({
        nrCpf: cpf,
        nmUsuario: cpf
      });
      usuario = await this.usuarioRepository.save(usuario);
    }

    let laudo = await this.laudoRepository.findOne({
      where: { nrLpco: lpco, usuario: { id: usuario.id } }
    });

    // Se forceReprocess for false e já existir dados processados, retorna o cache
    if (!forceReprocess && laudo?.jsDadoBruto) {
      this.logger.log(`[${new Date().toISOString()}] Retornando dados do cache - LPCO: ${lpco}`);
      return {
        nrLpco: laudo.nrLpco,
        jsDadoBruto: laudo.jsDadoBruto,
        obLaudoUsuario: laudo.obLaudoUsuario ?? ''
      };
    }

    const usuarioFileName = generateFileName(lpco, 'usuario');

    const formData = new FormData();
    formData.append('laudo', laudoFile.buffer, {
      filename: laudoFile.originalname,
      contentType: laudoFile.mimetype,
      knownLength: laudoFile.size
    });
    formData.append('lpco', lpco);

    try {
      this.logger.log(
        `[${new Date().toISOString()}] Iniciando upload para MinIO - Arquivo: ${usuarioFileName}`
      );
      await minioClient.putObject(MINIO_BUCKET, usuarioFileName, laudoFile.buffer, laudoFile.size, {
        'Content-Type': laudoFile.mimetype
      });
      this.logger.log(`[${new Date().toISOString()}] Upload para MinIO concluído`);

      this.logger.log(`[${new Date().toISOString()}] Iniciando processamento IA - LPCO: ${lpco}`);
      const apiResponse = await iaApi.post('/extract-pdf-data', formData, {
        headers: {
          ...formData.getHeaders()
        },
        maxBodyLength: Infinity,
        maxContentLength: Infinity,
        timeout: 300000, // 5 minutos
        onUploadProgress: progressEvent => {
          const percent = Math.round((progressEvent.loaded * 100) / (progressEvent.total || 1));
          this.logger.log(`[${new Date().toISOString()}] Upload progress: ${percent}%`);
        }
      });
      this.logger.log(`[${new Date().toISOString()}] Processamento IA concluído`);

      if (!laudo) {
        laudo = this.laudoRepository.create({
          nrLpco: lpco,
          jsDadoBruto: apiResponse.data,
          obLaudoUsuario: usuarioFileName,
          usuario
        });
      } else {
        laudo.jsDadoBruto = apiResponse.data;
        laudo.obLaudoUsuario = usuarioFileName;
      }

      const savedLaudo = await this.laudoRepository.save(laudo);
      this.logger.log(`[${new Date().toISOString()}] Laudo salvo no banco de dados`);

      return {
        nrLpco: savedLaudo.nrLpco,
        jsDadoBruto: savedLaudo.jsDadoBruto,
        obLaudoUsuario: savedLaudo.obLaudoUsuario ?? ''
      };
    } catch (error: any) {
      this.logger.error(`[${new Date().toISOString()}] Erro no processamento do laudo:`, error);

      // Log específico para timeouts
      if (error.code === 'ECONNABORTED' || error.message?.includes('timeout')) {
        this.logger.error('Timeout detectado - processamento demorou mais que 5 minutos');
        throw new BadRequestException(
          'Processamento demorou mais que o esperado. Tente novamente ou use um arquivo menor.'
        );
      }

      throw new BadRequestException(
        error.response?.data || error.message || 'Erro ao processar a requisição'
      );
    }
  }
}

              'Content-Type': 'application/pdf'
            });
            laudo.obLaudoIa = fileName;
          } catch (e) {
            this.logger.error('Erro ao salvar arquivo no MinIO:', e);
            throw new BadRequestException('Erro ao salvar arquivo no MinIO');
          }
        }

        // Se recebeu dados extraídos, atualizar jsDadoBruto
        if (input.dados_extraidos) {
          laudo.jsDadoBruto = input.dados_extraidos;
        }

        // Limpar erro anterior se houver
        laudo.erroIa = undefined;
      } else if (input.status === 'ERRO') {
        laudo.status = LaudoStatus.ERRO;
        laudo.erroIa = input.mensagem || 'Erro desconhecido no processamento da IA';
      }

      await this.laudoRepository.save(laudo);

      return {
        message: 'Callback processado com sucesso',
        laudo_id: laudo.id,
        status: laudo.status
      };
    } catch (error: any) {
      laudo.status = LaudoStatus.ERRO;
      laudo.erroIa = error.message || 'Erro ao processar callback';
      await this.laudoRepository.save(laudo);

      throw new BadRequestException('Erro ao processar callback: ' + error.message);
    }
  }
}

@Injectable()
export class ProcessExtractAsyncService {
  private readonly logger = new Logger(ProcessExtractAsyncService.name);

  constructor(
    @Inject(LAUDO_REPOSITORY)
    private laudoRepository: Repository<LaudoOlivia>
  ) {}

  async dispararExtracao(laudo: LaudoOlivia, pdfBuffer: Buffer, fileName: string): Promise<void> {
    try {
      // 1. Salvar PDF no MinIO
      await minioClient.putObject(MINIO_BUCKET, fileName, pdfBuffer, pdfBuffer.length, {
        'Content-Type': 'application/pdf'
      });

      laudo.obLaudoUsuario = fileName;

      // 2. Atualizar status para PROCESSANDO
      laudo.status = LaudoStatus.PROCESSANDO;
      laudo.erroIa = undefined;
      await this.laudoRepository.save(laudo);

      // 3. Montar URL de callback
      const apiUrl = process.env.API_URL;
      const callbackUrl = `${apiUrl}/api/laudos/extract-callback/${laudo.id}`;
      this.logger.log(`process-extract-async.service - callback url ${callbackUrl}`);

      // 4. Preparar FormData para enviar à IA
      const formData = new FormData();
      formData.append('job_id', laudo.id.toString());
      formData.append('callback_url', callbackUrl);
      formData.append('lpco', laudo.nrLpco);
      formData.append('laudo', pdfBuffer, {
        filename: fileName,
        contentType: 'application/pdf'
      });

      // 5. Fazer requisição POST para a API da IA (FastAPI)
      await iaApi.post('/extract-pdf-data-async', formData, {
        headers: {
          ...formData.getHeaders()
        },
        maxBodyLength: Infinity,
        maxContentLength: Infinity
      });
    } catch (error: any) {
      // Em caso de erro ao disparar, marcar como ERRO
      laudo.status = LaudoStatus.ERRO;
      laudo.erroIa =
        error.response?.data?.detail ||
        error.response?.data?.message ||
        error.message ||
        'Erro ao disparar extração na IA';
      await this.laudoRepository.save(laudo);
      this.logger.error(
        `Erro ao disparar extração IA para laudo ${laudo.id}:`,
        error?.stack || error
      );
    }
  }
}

export async function removeFile(filePath?: string): Promise<void> {
  if (!filePath) return;
  try {
    await fs.promises.unlink(filePath);
  } catch (error: any) {
    if (error.code !== 'ENOENT') {
      logger.error(`Erro ao remover o arquivo ${filePath}:`, error);
      return;
    }
  }

  try {
    await fs.promises.access(filePath);
    logger.warn(`Arquivo ${filePath} ainda existe após tentativa de exclusão.`);
  } catch (error: any) {
    if (error.code !== 'ENOENT') {
      logger.error(`Erro ao verificar a exclusão do arquivo ${filePath}:`, error);
    }
  }
}

export function extractCpf(sub?: string): string {
  if (!sub || typeof sub !== 'string') {
    throw new BadRequestException("O campo 'sub' (CPF) é obrigatório.");
  }
  const cpf = sub.replace(/\D/g, '');
  if (!cpf) {
    throw new BadRequestException("O campo 'sub' deve conter um CPF válido.");
  }
  return cpf;
}

  return `${safeLpco}_${type}_${hash}.pdf`;

const escapeRegex = (value: string) => value.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');

const buildOriginMatcher = (rawOrigin: string) => {
  const origin = rawOrigin.trim();

  if (!origin) {
    return null;
  }

  if (origin === '*') {
    return () => true;
  }

  if (origin.includes('*')) {
    const protocolSeparatorIndex = origin.indexOf('://');
    const hasPort =
      protocolSeparatorIndex >= 0
        ? origin.indexOf(':', protocolSeparatorIndex + 3) !== -1
        : origin.includes(':');
    const pattern = escapeRegex(origin).replace(/\\\*/g, '.*');
    const finalPattern = hasPort ? pattern : `${pattern}(?::\\d+)?`;
    const regex = new RegExp(`^${finalPattern}$`, 'i');
    return (candidate: string) => regex.test(candidate);
  }

  const normalized = origin.toLowerCase();
  return (candidate: string) => candidate === normalized;
};

async function bootstrap() {
  const app = await NestFactory.create(AppModule, {
    bodyParser: false
  });

  const logger = app.get(AppLogger);

  const originMatchers = DEFAULT_ALLOWED_ORIGINS.map(buildOriginMatcher).filter(
    (matcher): matcher is (origin: string) => boolean => matcher !== null
  );

  const isOriginAllowed = (origin: string | undefined) => {
    if (!origin) {
      logger.log('Origin is undefined, allowing request', 'CORS');
      return true;
    }

    const normalizedOrigin = origin.toLowerCase();
    const allowed = originMatchers.some(matcher => matcher(normalizedOrigin));

    if (allowed) {
      logger.log(`Origin allowed: ${origin}`, 'CORS');
    } else {
      logger.warn(`Origin not allowed by CORS: ${origin}`, 'CORS');
    }

    return allowed;
  };

  app.enableCors({
    origin: (
      origin: string | undefined,
      callback: (err: Error | null, allow: boolean | string) => void
    ) => {
      if (isOriginAllowed(origin)) {
        callback(null, true);
      } else {
        callback(new Error(`Origin not allowed by CORS: ${origin ?? '<unknown>'}`), false);
      }
    },
    credentials: true,
    methods: ['GET', 'HEAD', 'PUT', 'PATCH', 'POST', 'DELETE', 'OPTIONS'],

// Gera um sessionId inicial para ser usado antes do login completo
const generateSessionId = () => {
  return 'sess_' + Date.now() + '_' + Math.random().toString(36).substring(2, 9);
};

const fallbackSessionId = generateSessionId();

const resolveSessionId = () => {
  if (typeof window === 'undefined') {
    return fallbackSessionId;
  }

  const stored = sessionStorage.getItem('session_id');
  if (stored && stored.trim()) {
    return stored.trim();
  }

  return fallbackSessionId;
};

    }

    if (typeof window !== 'undefined') {

      }
    }

    return config;
  },
  error => Promise.reject(error)

export function scheduleAutoLogout(expUnix, onLogout, skewSec = 60) {
  if (!expUnix) return;
  const msUntil = expUnix * 1000 - Date.now() - skewSec * 1000;
  if (msUntil <= 0) {
    onLogout();
    return;
  }
  if (window.__logoutTimer) clearTimeout(window.__logoutTimer);
  window.__logoutTimer = setTimeout(onLogout, msUntil);
}

const getFilenameFromDisposition = header => {
  if (!header) return null;
  const match = /filename\*=UTF-8''([^;]+)|filename="?([^";]+)"?/i.exec(header);
  if (!match) return null;
  return decodeURIComponent(match[1] || match[2]).trim();
};

export const downloadLaudo = async laudoId => {
  if (!laudoId) {
    throw new Error('Identificador do laudo inválido.');
  }

  const response = await apiClient.get(`/api/laudo/${laudoId}`, {
    responseType: 'blob'
  });

  const contentType = response.headers['content-type'] || 'application/pdf';
  const blob = new Blob([response.data], { type: contentType });
  const objectUrl = URL.createObjectURL(blob);
  const link = document.createElement('a');

  const filename =
    getFilenameFromDisposition(response.headers['content-disposition']) || `${laudoId}.pdf`;

  link.href = objectUrl;
  link.download = filename;
  link.style.display = 'none';
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
  URL.revokeObjectURL(objectUrl);
};

/**

 * @param {number} laudoId - ID do laudo
 * @returns {Promise<{id: number, status: string, erroIa?: string, obLaudoIa?: string, nrLpco: string}>}

export const checkLaudoStatus = async laudoId => {
  if (!laudoId) {
    throw new Error('ID do laudo inválido');
  }

  const response = await apiClient.get(`/api/laudos/status/${laudoId}`);
  return response.data;

/**
 * Inicia polling para verificar status do laudo
 * @param {number} laudoId - ID do laudo
 * @param {Function} onStatusChange - Callback chamado quando status muda
 * @param {number} interval - Intervalo em ms (padrão: 5000)
 * @returns {Function} Função para parar o polling
 */
export const startLaudoPolling = (laudoId, onStatusChange, interval = 5000) => {
  let timeoutId = null;
  let isPolling = true;

  const poll = async () => {
    if (!isPolling) return;

    try {
      const status = await checkLaudoStatus(laudoId);
      onStatusChange(status);

      // Continuar polling apenas se status for PENDENTE ou PROCESSANDO
      if (status.status === 'PENDENTE' || status.status === 'PROCESSANDO') {
        timeoutId = setTimeout(poll, interval);
      }
    } catch (error) {
      console.error('Erro ao verificar status do laudo:', error);
      // Em caso de erro, tentar novamente após o intervalo
      timeoutId = setTimeout(poll, interval);
    }
  };

  // Iniciar polling
  poll();

  // Retornar função para parar o polling
  return () => {
    isPolling = false;
    if (timeoutId) {
      clearTimeout(timeoutId);
    }
  };
};

    id => {
      stopPolling();

        checkJobStatus(id);

      stopPolling();

const formatDate = isoString => {
  if (!isoString) return 'Data inválida';

  const date = new Date(isoString);

  if (Number.isNaN(date.getTime())) return 'Data inválida';

  const formatted = new Intl.DateTimeFormat('pt-BR', {
    day: '2-digit',
    month: '2-digit',
    year: 'numeric',
    hour: '2-digit',
    minute: '2-digit',
    hour12: false,
    timeZone: 'America/Sao_Paulo'
  }).format(date);

  const [datePart, timePart] = formatted.split(', ');
  const [hour, minute] = timePart.split(':');
  return `${datePart} - ${hour}h${minute}`;
};

const Processamento = () => {
  const breadcrumbs = [{ label: 'Inicio', href: '/arquivos' }, { label: 'Processamento' }];

  const router = useRouter();
  const queryClient = useQueryClient();

  const laudo = queryClient.getQueryData(['laudo']);

  const [formData, setFormData] = useState({});
  const [pdfData, setPdfData] = useState(null);
  const [validationError, setValidationError] = useState('');
  const responseApi = queryClient.getQueryData(['extract-pdf-data']);

  const cloneData = data => (data ? JSON.parse(JSON.stringify(data)) : {});

  // Redireciona APENAS na montagem inicial se não houver dados
  const hasInitialData = React.useRef(!!(laudo && responseApi));
  useEffect(() => {
    if (!hasInitialData.current) {
      router.push('/arquivos');
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  useEffect(() => {
    if (responseApi?.jsDadoBruto) {
      const initialData = cloneData(responseApi.jsDadoBruto);

      // Função para renomear campos obrigatórios para as chaves corretas
      const normalizeCabecalhoKeys = cabecalho => {
        if (!cabecalho) return cabecalho;
        const renamed = { ...cabecalho };
        // Lote
        if ('Lote' in renamed && !('Número Lote' in renamed)) {
          renamed['Número Lote'] = renamed['Lote'];
          delete renamed['Lote'];
        }
        // Contratante
        if ('Contratante' in renamed && !('Cliente' in renamed)) {
          renamed['Cliente'] = renamed['Contratante'];
          delete renamed['Contratante'];
        }
        return renamed;
      };

      const normalizeQuimicoKeys = quimico => {
        if (!quimico) return quimico;
        const renamed = { ...quimico };
        // Grupo
        if ('Grupo' in renamed && !('GRUPO' in renamed)) {
          renamed['GRUPO'] = renamed['Grupo'];
          delete renamed['Grupo'];
        }
        return renamed;
      };

      // Remove apenas campos internos (prefixo _), mantém campos vazios para preenchimento manual
      const removeInternalFields = obj => {
        if (!obj) return obj;
        const cleaned = {};
        Object.entries(obj).forEach(([key, value]) => {
          if (!key.startsWith('_')) {
            cleaned[key] = value ?? '';
          }
        });
        return cleaned;
      };

      const filteredData = {
        ...initialData,
        cabecalho: {
          ...removeInternalFields(normalizeCabecalhoKeys(initialData.cabecalho)),
          // Finalidade sempre inicia vazio - deve ser sel
          // selecionado pelo usuário
          finalidade_classificacao: ''
        },
        quimico: removeInternalFields(normalizeQuimicoKeys(initialData.quimico))
      };

      setFormData(cloneData(filteredData));
    }
  }, [responseApi]);

  // Processar o PDF base64
  useEffect(() => {
    if (laudo) {
      try {
        let pdfDataToUse = null;

        // Se já é um data URL válido
        if (laudo.startsWith('data:application/pdf;base64,')) {
          pdfDataToUse = laudo;
        }
        // Se é apenas base64 sem o prefixo
        else if (typeof laudo === 'string' && laudo.length > 100) {
          pdfDataToUse = `data:application/pdf;base64,${laudo}`;
        }
        // Se é outro formato
        else {
          pdfDataToUse = laudo;
        }
        setPdfData(pdfDataToUse);
      } catch (error) {
        console.error('Erro ao processar laudo:', error);
      }
    }
  }, [laudo]);

  const mutation = useMutation({
    mutationFn: async () => {
      const body = {
        lpco: responseApi.nrLpco,
        chem_data: formData
      };

      const { data } = await apiClient.post('/api/data-processing', body);
      return data;
    },

    onSuccess: data => {
      queryClient.setQueryData(['data-processing'], data);
      router.push('/resultado');
    },

    onError: error => {
      console.error('Erro ao salvar dados:', error);
    }
  });

  const handleSave = () => {
    // Validação dos campos obrigatórios do cabeçalho

/**
 * Hook para fazer polling do status de um laudo
 * @param {number|null} laudoId - ID do laudo para fazer polling
 * @param {Object} options - Opções de configuração
 * @param {number} options.interval - Intervalo em ms (padrão: 5000)
 * @param {boolean} options.enabled - Se o polling está habilitado (padrão: true)
 * @returns {Object} Estado do polling
 */
export const useLaudoPolling = (laudoId, options = {}) => {
  const { interval = 5000, enabled = true } = options;

  const [status, setStatus] = useState(null);
  const [isPolling, setIsPolling] = useState(false);
  const [error, setError] = useState(null);
  const stopPollingRef = useRef(null);

  const handleStatusChange = useCallback(newStatus => {
    setStatus(newStatus);

    // Se chegou a um estado final, parar o polling
    if (newStatus.status === 'CONCLUIDO' || newStatus.status === 'ERRO') {
      setIsPolling(false);
    }
  }, []);

  useEffect(() => {
    // Se não tiver laudoId ou não estiver habilitado, não fazer nada
    if (!laudoId || !enabled) {
      return;
    }

    // Se já chegou a um estado final, não iniciar polling
    if (status?.status === 'CONCLUIDO' || status?.status === 'ERRO') {
      return;
    }

    setIsPolling(true);
    setError(null);

    // Iniciar polling
    stopPollingRef.current = startLaudoPolling(laudoId, handleStatusChange, interval);

    // Cleanup: parar polling quando componente desmontar
    return () => {
      if (stopPollingRef.current) {
        stopPollingRef.current();
        stopPollingRef.current = null;
      }
      setIsPolling(false);
    };
  }, [laudoId, enabled, interval, handleStatusChange, status?.status]);

  return {
    status,
    isPolling,
    error,
    isPending: status?.status === 'PENDENTE',
    isProcessing: status?.status === 'PROCESSANDO',
    isCompleted: status?.status === 'CONCLUIDO',
    isError: status?.status === 'ERRO'
  };
};

const devUserSession = {

export const AuthProvider = ({ children }) => {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [authError, setAuthError] = useState(null);
  const [isLoggingOut, setIsLoggingOut] = useState(false);
  const router = useRouter();

  const startDevSession = useCallback(async () => {
    sessionStorage.setItem('access_token', devUserSession.accessToken);
    sessionStorage.setItem('session_id', devUserSession.sessionId);
    sessionStorage.setItem('expires_at', String(devUserSession.expiresAt()));

    // Buscar status real do usuário do backend
    try {
      const { data } = await apiClient.get('/api/usuarios/me/status');

      const userWithStatus = {
        sub: devUserSession.user.sub,
        name: devUserSession.user.name,
        email: devUserSession.user.email,
        email_verified: devUserSession.user.email_verified,
        picture: devUserSession.user.picture,
        role: data.role,
        aprovado: data.aprovado,
        pendente: data.pendente
      };

      sessionStorage.setItem('user', JSON.stringify(userWithStatus));
      setUser(userWithStatus);

      // Se não estiver aprovado, redirecionar para pendente
      if (!data.aprovado || data.pendente) {
        router.push('/pendente');
      } else {
        router.push('/arquivos');
      }
    } catch (error) {
      console.error('Erro ao buscar status do usuário dev:', error);
      // Fallback para dados padrão
      sessionStorage.setItem('user', JSON.stringify(devUserSession.user));
      setUser(devUserSession.user);
      router.push('/arquivos');
    }
  }, [router]);

  useEffect(() => {
    const initAuth = async () => {
      const savedUser = sessionStorage.getItem('user');
      if (savedUser) {
        setUser(JSON.parse(savedUser));
      } else if (isDevAuthBypassEnabled) {
        await startDevSession();
      }
      setLoading(false);
    };

    initAuth();
  }, [startDevSession]);

  const loginWithGovBr = async () => {
    try {
      if (isDevAuthBypassEnabled) {
        await startDevSession();
        return;
      }
      window.location.href = `${process.env.NEXT_PUBLIC_BACKEND_URL}/auth/govbr/login`;
    } catch (error) {
      console.error('Falha ao iniciar login Gov.br:', error?.message || error);
    }
  };

  const handleGovBrCallback = useCallback(
    async (code, state) => {
      try {
        if (isDevAuthBypassEnabled) {
          await startDevSession();
          return;
        }
        const { data } = await apiClient.get('/auth/govbr/callback', {
          params: { code, state }
        });

        if (!data?.access_token) throw new Error(data?.message || 'Falha no login');

        sessionStorage.setItem('access_token', data.access_token);
        if (data.session_id) sessionStorage.setItem('session_id', data.session_id);
        if (data.expires_at) sessionStorage.setItem('expires_at', String(data.expires_at));

        const userUi = {
          sub: data.user.sub,
          name: data.user.name ?? null,
          email: data.user.email ?? null,
          email_verified: !!data.user.email_verified,
          picture: data.user.picture ?? null,
          role: data.role ?? 'FISCAL',
          aprovado: data.aprovado ?? false,
          pendente: data.pendente ?? false
        };
        sessionStorage.setItem('user', JSON.stringify(userUi));
        setUser(userUi);

        // Se o usuário está pendente de aprovação, redireciona para página de espera
        if (data.pendente || !data.aprovado) {
          router.push('/pendente');
          return;
        }

        if (data.expires_at) {
          scheduleAutoLogout(
            Math.floor(data.expires_at / 1000),
            () => {
              sessionStorage.clear();
              setUser(null);
              router.replace('/');
            },
            120
          );
        }

        router.push('/arquivos');
      } catch (error) {
        console.error('Callback falhou:', error?.message || error);
        const msg =
          (error && error.response && error.response.data && error.response.data.message) ||
          error?.message ||
          'Falha no login';
        setAuthError(typeof msg === 'string' ? msg : JSON.stringify(msg));
        router.replace('/');
      }
    },
    [router, startDevSession]
  );

  const checkUserStatus = useCallback(async () => {
    try {
      const { data } = await apiClient.get('/api/usuarios/me/status');

      const savedUser = sessionStorage.getItem('user');
      if (savedUser) {
        const userObj = JSON.parse(savedUser);
        const updatedUser = {
          ...userObj,
          aprovado: data.aprovado,
          role: data.role,
          pendente: data.pendente
        };
        sessionStorage.setItem('user', JSON.stringify(updatedUser));
        setUser(updatedUser);
      }

      return data;
    } catch (error) {
      console.error('Falha ao verificar status do usuário:', error?.message || error);
      return null;
    }
  }, []);

  const logout = async () => {
    try {
      setIsLoggingOut(true);
      const sessionId = sessionStorage.getItem('session_id') || '';

      if (isDevAuthBypassEnabled) {
        sessionStorage.clear();
        setUser(null);
        router.push('/');
        return;
      }

      try {
        await apiClient.get('/auth/govbr/logout', {
          params: { session_id: sessionId }
        });
      } catch (err) {
        console.warn('Falha ao notificar logout no backend:', err?.message || err);
      }
      sessionStorage.clear();
      setUser(null);

      const defaultPostLogoutPath = '/auth/govbr/sair';
      const redirectUri =
        process.env.NEXT_PUBLIC_LOGOUT_REDIRECT_URI ||
        (typeof window !== 'undefined'
          ? `${window.location.origin}${defaultPostLogoutPath}`
          : defaultPostLogoutPath);

      const ssoUrl = process.env.NEXT_PUBLIC_SSO_URL || process.env.NEXT_PUBLIC_GOVBR_URL_PROVIDER;

      if (!ssoUrl || ssoUrl === 'undefined') {
        console.error(
          'NEXT_PUBLIC_SSO_URL e NEXT_PUBLIC_GOVBR_URL_PROVIDER não estão definidas. Redirecionando para a página inicial.'
        );
        router.push('/');
        return;
      }

      const ssoLogout = `${ssoUrl}/logout?post_logout_redirect_uri=${encodeURIComponent(
        redirectUri
      )}`;

      console.log('Redirecionando para logout SSO:', ssoLogout);

      if (typeof window !== 'undefined') {
        window.location.href = ssoLogout;
      } else {
        router.push('/');
      }
    } catch (e) {
      console.error('Erro durante logout:', e);
      setIsLoggingOut(false);
      router.push('/');
    }
  };

  return (
    <AuthContext.Provider
      value={{
        user,
        loginWithGovBr,
        handleGovBrCallback,
        logout,
        authError,
        clearAuthError: () => setAuthError(null),
        isAuthenticated: !!user,
        loading,
        isLoggingOut,
        checkUserStatus
      }}>
      {children}
    </AuthContext.Provider>
  );
};

const MaskedInput = ({ value, onChange }) => {
  const handleChange = e => {
    let cleanedValue = e.target.value;

    if (cleanedValue.endsWith('-')) {
      cleanedValue = cleanedValue.slice(0, -1);
    }

    const newEvent = {
      ...e,
      target: {
        ...e.target,
        value: cleanedValue
      }
    };

    onChange(newEvent);
  };

  return (
    <InputMask
      mask='* 99 99999999-**'
      value={value}
      onChange={handleChange}
      alwaysShowMask={false}>
      {inputProps => (
        <TextField
          {...inputProps}
          label='LPCO'
          variant='outlined'
          fullWidth
        />
      )}
    </InputMask>
  );
};

const PrivateRoute = ({ children, requireAdmin = false }) => {
  const { isAuthenticated, loading, isLoggingOut, user } = useAuth();
  const router = useRouter();

  useEffect(() => {
    if (!loading && !isAuthenticated && !isLoggingOut) {
      router.push('/');
      return;
    }

    if (!loading && isAuthenticated && user) {
      // Se usuário não está aprovado, redireciona para página de pendente
      if (!user.aprovado) {
        if (router.pathname !== '/pendente') {
          router.push('/pendente');
        }
        return;
      }

      // Se a rota requer admin e o usuário não é admin, redireciona para /arquivos
      if (requireAdmin && user.role !== 'ADMIN') {
        if (router.pathname !== '/arquivos') {
          router.push('/arquivos');
        }
        return;
      }
    }
  }, [isAuthenticated, loading, isLoggingOut, user, requireAdmin, router]);

  if (loading) return null;
  if (!isAuthenticated) return null;

  // Se usuário não está aprovado, só renderiza se estiver na página /pendente
  if (user && !user.aprovado) {
    if (router.pathname !== '/pendente') return null;
  }

  // Se requer admin e usuário não é admin, não renderiza
  if (requireAdmin && user && user.role !== 'ADMIN') {
    return null;
  }

  return children;
};

  it('should return a string of 11 digits', () => {
    const cpf = generateCPF();
    expect(typeof cpf).toBe('string');
    expect(cpf).toHaveLength(11);
    expect(/^\d{11}$/.test(cpf)).toBe(true);
  });

  it('should generate a valid CPF with correct verifier digits', () => {
    const cpf = generateCPF();
    const digits = cpf.split('').map(Number);
    const firstNine = digits.slice(0, 9);

    const sum1 = firstNine.reduce((acc, d, i) => acc + d * (10 - i), 0);
    const remainder1 = sum1 % 11;
    const expectedFirstVerifier = remainder1 < 2 ? 0 : 11 - remainder1;

    const sum2 = firstNine.reduce((acc, d, i) => acc + d * (11 - i), 0) + expectedFirstVerifier * 2;
    const remainder2 = sum2 % 11;
    const expectedSecondVerifier = remainder2 < 2 ? 0 : 11 - remainder2;

    expect(digits[9]).toBe(expectedFirstVerifier);
    expect(digits[10]).toBe(expectedSecondVerifier);
  });
});
export function generateCPF(): string {
  const randomDigit = (): number => Math.floor(Math.random() * 10);
  const digits: number[] = Array.from({ length: 9 }, randomDigit);

  let sum = digits.reduce((acc, digit, index) => acc + digit * (10 - index), 0);
  let remainder = sum % 11;
  const firstVerifier = remainder < 2 ? 0 : 11 - remainder;

  sum = digits.reduce((acc, digit, index) => acc + digit * (11 - index), 0) + firstVerifier * 2;
  remainder = sum % 11;
  const secondVerifier = remainder < 2 ? 0 : 11 - remainder;

  const cpf = [...digits, firstVerifier, secondVerifier].join('');
  return cpf;
}

export function base64url(input: Buffer | string): string {
  const buff = Buffer.isBuffer(input) ? input : Buffer.from(input);
  return buff.toString('base64').replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/g, '');
}

export function generateCodeVerifier(length = 64): string {
  return base64url(randomBytes(length));

export function deriveCodeChallengeS256(codeVerifier: string): string {
  const hash = createHash('sha256').update(codeVerifier).digest();
  return base64url(hash);

export class FakeUsersRepository {
  public usersRepository: User[] = [];

  save(user: User) {
    this.usersRepository.push(user);
    return user;
  }

  findOneBy(query: queryRequest) {
    if (query.id) {
      return this.usersRepository.find(user => user.id.toString() === query.id?.toString());
    } else if (query.email) {
      return this.usersRepository.find(user => user.email === query.email);
    }
    return undefined;
  }

  async findOne(options: {
    where: Array<{ id?: string; cpf?: string }>;
    relations: string[];
  }): Promise<User | undefined> {
    const user = this.usersRepository.find(user =>
      options.where.some(condition => {
        if (condition.id !== undefined && user.id === condition.id) return true;
        if (condition.cpf !== undefined && user.cpf === condition.cpf) return true;
        return false;
      })
    );

    if (user) {
      user.role = { id: 1, name: 'admin' } as Role;
    }

    return user;
  }

  remove(user: User): User | undefined {
    const index = this.usersRepository.findIndex(
      storedUser => storedUser.id.toString() === user.id.toString()
    );

    if (index !== -1) {
      const [removedUser] = this.usersRepository.splice(index, 1);
      return removedUser;
    }

    return undefined;
  }

  update(id: string, partialUser: Partial<User>): User | undefined {
    const index = this.usersRepository.findIndex(user => user.id.toString() === id.toString());

    if (index !== -1) {
      const updatedUser = {
        ...this.usersRepository[index],
        ...partialUser
      };
      this.usersRepository[index] = updatedUser as User;
      return updatedUser as User;
    }

    return undefined;
  }
}

let dataSourceInstance: DataSource | null = null;

export const databaseProviders = [
  {
    provide: MAPA_POSTGRES_DATA_SOURCE,
    imports: [EnvironmentModule],
    inject: [EnvironmentService, AppLogger],
    useFactory: async (
      configService: EnvironmentService,
      logger: AppLogger
    ): Promise<DataSource> => {
      if (dataSourceInstance) {
        logger.log('Retornando DataSource já inicializado.', 'databaseProviders');
        return dataSourceInstance;
      }

      logger.log('Inicializando DataSource.', 'databaseProviders');
      const dataSource = new DataSource({
        type: configService.get('MAPA_TYPEORM_POSTGRES_TYPE'),
        host: configService.get('POSTGRES_HOST'),
        schema: configService.get('POSTGRES_SCHEMA'),
        port: configService.get('POSTGRES_PORT'),
        username: configService.get('POSTGRES_USER'),
        password: configService.get('POSTGRES_PASSWORD'),
        database: configService.get('POSTGRES_DB'),
        entities: [...entidades],
        synchronize: false,
        logging: configService.get('POSTGRES_LOGAR'),
        extra: {
          timezone: 'UTC'
        }
      });

      try {
        await dataSource.initialize();
        logger.log('Retornando DataSource já inicializado.', 'databaseProviders');
        dataSourceInstance = dataSource;
        return dataSource;
      } catch (error) {
        if (error instanceof Error) {
          logger.error(
            `Erro ao inicializar DataSource: ${error.message}`,
            error.stack,
            'databaseProviders'
          );
        } else {
          logger.error('Erro desconhecido ao inicializar DataSource', '', 'databaseProviders');
        }
        throw error;
      }
    }
  }
];

export const getDataSource = (): DataSource => {
  if (!dataSourceInstance) {
    throw new Error('O DataSource ainda não foi inicializado!');
  }
  return dataSourceInstance;
};

interface ExtractPdfDataBody {
  lpco: string;
  sub: string;
  forceReprocess?: string;
}

const bodySchema = z
  .object({
    lpco: z.string(),
    sub: z.string(),
    forceReprocess: z.string().optional()
  })
  .strict();

const bodyPipe = new ZodValidationPipe(bodySchema);

@Controller('api')
export class ExtractPdfDataController {
  private readonly logger = new Logger(ExtractPdfDataController.name);

  constructor(
    private createExtractJobService: CreateExtractJobService,
    private processExtractAsyncService: ProcessExtractAsyncService,
    @Inject(USUARIO_REPOSITORY)
    private usuarioRepository: Repository<Usuario>
  ) {}

  @Post('extract-pdf-data')
  @HttpCode(HttpStatus.ACCEPTED)
  @UseInterceptors(
    FileInterceptor('laudo', {
      storage: memoryStorage(),
      limits: { fileSize: 5 * 1024 * 1024 },
      fileFilter: (_, file, cb) => {
        if (file.mimetype === 'application/pdf') {
          cb(null, true);
        } else {
          cb(new BadRequestException('Somente PDF'), false);
        }
      }
    })
  )
  async execute(
    @UploadedFile() laudoFile: Express.Multer.File,
    @Body(bodyPipe) { lpco, sub, forceReprocess }: ExtractPdfDataBody
  ) {
    this.logger.log(
      `Requisição de extração recebida - LPCO: ${lpco}, forceReprocess: ${forceReprocess || 'false'}`
    );

    if (!laudoFile) {
      this.logger.warn(`Requisição rejeitada - Arquivo não enviado - LPCO: ${lpco}`);
      throw new BadRequestException("Arquivo 'laudo' não foi enviado.");
    }

    this.logger.log(
      `Arquivo recebido - Nome: ${laudoFile.originalname}, Tamanho: ${(laudoFile.size / 1024).toFixed(2)} KB, LPCO: ${lpco}`
    );

    // Buscar ou criar usuário
    const cpf = extractCpf(sub);
    this.logger.log(`Processando usuário - CPF: ${cpf.substring(0, 3)}***, LPCO: ${lpco}`);

    let usuario = await this.usuarioRepository.findOne({ where: { nrCpf: cpf } });

    if (!usuario) {
      this.logger.log(`Criando novo usuário - CPF: ${cpf.substring(0, 3)}***, LPCO: ${lpco}`);
      usuario = this.usuarioRepository.create({
        nrCpf: cpf,
        nmUsuario: cpf
      });
      usuario = await this.usuarioRepository.save(usuario);
      this.logger.log(`Usuário criado com sucesso - ID: ${usuario.id}, LPCO: ${lpco}`);
    } else {
      this.logger.log(`Usuário encontrado - ID: ${usuario.id}, LPCO: ${lpco}`);
    }

    // Criar job de extração
    this.logger.log(`Criando job de extração - LPCO: ${lpco}`);
    const job = await this.createExtractJobService.execute({
      lpco,
      laudoFile,
      usuario,
      forceReprocess: forceReprocess === 'true'
    });
    this.logger.log(`Job criado - ID: ${job.id}, Status: ${job.status}, LPCO: ${lpco}`);

    // Se retornou dados do cache, retornar imediatamente
    if (job.cached) {
      this.logger.log(`Retornando dados do cache - Job ID: ${job.id}, LPCO: ${lpco}`);
      return {
        id: job.id,
        status: job.status,
        cached: true,
        jsDadoBruto: job.jsDadoBruto,
        obLaudoUsuario: job.obLaudoUsuario,
        nrLpco: job.nrLpco
      };
    }

    // Gerar nome do arquivo
    const fileName = generateFileName(lpco, 'usuario');
    this.logger.log(
      `Iniciando processamento assíncrono - Job ID: ${job.id}, Arquivo: ${fileName}, LPCO: ${lpco}`
    );

    // Disparar extração assíncrona (fire and forget)
    setImmediate(async () => {
      const laudo = await this.createExtractJobService['laudoRepository'].findOne({
        where: { id: job.id }
      });
      if (laudo) {
        this.logger.log(`Disparando extração assíncrona - Laudo ID: ${laudo.id}, LPCO: ${lpco}`);
        this.processExtractAsyncService
          .dispararExtracao(laudo, laudoFile.buffer, fileName)
          .catch(err => {
            this.logger.error(
              `Erro no processamento assíncrono de extração - Job ID: ${job.id}, LPCO: ${lpco}`,
              err.stack || err
            );
          });
      } else {
        this.logger.error(
          `Laudo não encontrado para processamento assíncrono - Job ID: ${job.id}, LPCO: ${lpco}`
        );
      }
    });

    // Retornar imediatamente com 202 Accepted
    this.logger.log(
      `Requisição aceita (202) - Job ID: ${job.id}, Status: ${job.status}, LPCO: ${lpco}`
    );
    return {
      id: job.id,
      status: job.status
    };
  }
}

@Injectable()
export class ErrorsInterceptor implements NestInterceptor {
  constructor(private readonly logger: AppLogger) {}
  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    const contextClass = context.getClass().name;
    const contextHandler = context.getHandler().name;
    const contextRef = `${contextClass}.${contextHandler}`;
    return next.handle().pipe(
      catchError(error => {
        if (error instanceof Error) {
          this.logger.error(
            `Erro em ${contextRef}: ${error.name}: ${error.message}`,
            error.stack,
            'ErrorsInterceptor'
          );
        } else {
          this.logger.error(
            'Erro desconhecido ocorrido no ErrorsInterceptor',
            '',
            'ErrorsInterceptor'
          );
        }

        if (error instanceof ZodError) {
          const formattedErrors = error.issues
            .map(err => `${err.path.join('.')} - ${err.message}`)
            .join(', ');
          return throwError(() => new InvalidFormatationException(`${formattedErrors}`));
        }

        if (error instanceof Error) {
          error.message = `${error.name}: ${error.message}`;
        }

        return throwError(() => error);
      })
    );
  }
}

type ExceptionResponse = string | { message: string; error: string };

@Catch()
export class AllExceptionsFilter implements ExceptionFilter {
  catch(exception: unknown, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();
    const request = ctx.getRequest<Request>();

    let status = HttpStatus.INTERNAL_SERVER_ERROR;
    let message = 'Internal server error';

    if (exception instanceof HttpException) {
      status = exception.getStatus();
      const exceptionResponse: ExceptionResponse = exception.getResponse() as ExceptionResponse;
      message =
        typeof exceptionResponse === 'string' ? exceptionResponse : exceptionResponse.message;
    } else if (exception instanceof Error) {
      message = exception.message;
    }

    response.status(status).json({
      statusCode: status,
      timestamp: new Date().toISOString(),
      path: request.url,
      message
    });
  }
}

export enum Action {
  Manage = 'manage',
  Create = 'create',
  Read = 'read',
  Update = 'update',
  Delete = 'delete',
  ManageAccount = 'manage-account',
  ManageAccountLinks = 'manage-account-links',
  ViewProfile = 'view-profile'
}

export type Subjects = 'all' | 'user' | any;
export type AppAbility = PureAbility<[Action, Subjects]>;

function isMatch(object: any, conditions: any): boolean {
  if (object === conditions) return true;
  if (
    typeof object !== 'object' ||
    object === null ||
    typeof conditions !== 'object' ||
    conditions === null
  ) {
    return false;
  }
  for (const key in conditions) {
    if (!Object.prototype.hasOwnProperty.call(object, key)) {
      return false;
    }
    if (!isMatch(object[key], conditions[key])) {
      return false;
    }
  }
  return true;
}

interface createForUserParam {
  payload: KeycloackPayload;
  resource: string;
}

export class CaslAbilityFactory {
  createForUser({ payload, resource }: createForUserParam): AppAbility {
    const { can, build } = new AbilityBuilder(PureAbility as AbilityClass<AppAbility>);

    const allowedActions: Action[] = [
      Action.ManageAccount,
      Action.ManageAccountLinks,
      Action.ViewProfile
    ];

    const roles = payload.resource_access?.account?.roles || [];
    roles.forEach((role: Action) => {
      if (allowedActions.includes(role)) {
        can(role, resource, { id: payload.sub });
      }
    });

    const customConditionsMatcher: ConditionsMatcher<any> = (conditions: any) => {
      return (object: any) => isMatch(object, conditions);
    };

    return build({
      conditionsMatcher: customConditionsMatcher,
      detectSubjectType: item => item.__type
    });
  }
}

interface Request {
  payload: KeycloackPayload;
  route: { path: string };
}

@Injectable()
export class PoliciesGuard implements CanActivate {
  constructor(
    private reflector: Reflector,
    private caslAbilityFactory: CaslAbilityFactory
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const policyHandlers =
      this.reflector.get<PolicyHandler[]>(CHECK_POLICIES_KEY, context.getHandler()) || [];

    const request = context.switchToHttp().getRequest<Request>();
    const routePath = request.route.path;
    const resource = getFirstWordBeforeParam(routePath);
    const payload = request.payload;
    const ability = this.caslAbilityFactory.createForUser({
      payload,
      resource
    });

    return policyHandlers.every(handler => {
      return this.execPolicyHandler(handler, ability, request);
    });
  }

  private execPolicyHandler(handler: PolicyHandler, ability: AppAbility, request: Request) {
    if (handler.length === 2) {
      return (handler as (a: AppAbility, r: Request) => boolean)(ability, request);
    }

    return (handler as (a: AppAbility) => boolean)(ability);
  }
}

@Injectable()
export class RoleGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.getAllAndOverride<UsuarioRole[]>(ROLES_KEY, [
      context.getHandler(),
      context.getClass()
    ]);

    if (!requiredRoles) {
      return true;
    }

    const request = context.switchToHttp().getRequest();
    const user: Usuario = request.user;

    if (!user) {
      throw new ForbiddenException('Usuário não autenticado');
    }

    if (!user.stAprovado) {
      throw new ForbiddenException('Usuário pendente de aprovação');
    }

    const hasRole = requiredRoles.some(role => user.cdRole === role);

    if (!hasRole) {
      throw new ForbiddenException('Você não tem permissão para acessar este recurso');
    }

    return true;
  }
}

function decodeJwt(jwt: string): any | null {
  try {
    const parts = jwt.split('.');
    if (parts.length < 2) return null;
    const payload = parts[1].replace(/-/g, '+').replace(/_/g, '/');
    const json = Buffer.from(payload, 'base64').toString('utf8');
    return JSON.parse(json);
  } catch (e) {
    console.error('Erro ao decodificar JWT:', e);
    return null;
  }
}

function decodeJwtHeader(jwt: string): any | null {
  try {
    const parts = jwt.split('.');
    if (parts.length < 1) return null;
    const headerB64 = parts[0].replace(/-/g, '+').replace(/_/g, '/');
    const json = Buffer.from(headerB64, 'base64').toString('utf8');
    return JSON.parse(json);
  } catch (e) {
    console.error('Erro ao decodificar header do JWT:', e);
    return null;
  }
}

@Injectable()
export class GovBrService {
  constructor(private readonly env: EnvironmentService) {}

  private generateState(): string {
    return randomBytes(16).toString('hex');
  }

  getAuthorizationUrl(): string {
    const baseUrl = this.env.get('GOVBR_URL');
    const authUrl = `${baseUrl}/authorize`;
    const clientId = this.env.get('GOVBR_CLIENT_ID');
    const redirectUri = this.env.get('GOVBR_REDIRECT_URL');
    const scope = 'openid email profile govbr_confiabilidades';

    const nonce = randomBytes(12).toString('hex');
    const state = this.generateState();
    const codeVerifier = generateCodeVerifier();
    const codeChallenge = deriveCodeChallengeS256(codeVerifier);
    const codeChallengeMethod = 'S256';

    PkceStore.set(state, codeVerifier);

    const params = new URLSearchParams({
      response_type: 'code',
      client_id: clientId,
      scope,
      redirect_uri: redirectUri
    });

    params.append('nonce', nonce);
    params.append('state', state);
    params.append('code_challenge', codeChallenge);
    params.append('code_challenge_method', codeChallengeMethod);

    const url = `${authUrl}?${params.toString()}`;
    return url;
  }

  async exchangeCode(code: string, state: string): Promise<any> {
    const baseUrl = this.env.get('GOVBR_URL');
    const tokenUrl = `${baseUrl}/token`;
    const clientId = this.env.get('GOVBR_CLIENT_ID');
    const clientSecret = this.env.get('GOVBR_CLIENT_SECRET');
    const redirectUri = this.env.get('GOVBR_REDIRECT_URL');
    const codeVerifier = PkceStore.get(state);
    if (!codeVerifier) {
      throw new BadRequestException('State inválido/expirado: code_verifier não encontrado');
    }

    const body = new URLSearchParams({
      grant_type: 'authorization_code',
      code,
      redirect_uri: redirectUri,
      client_id: clientId,
      code_verifier: codeVerifier
    });
    if (clientSecret) body.append('client_secret', clientSecret);

    try {
      const tokenRes = await axios.post(tokenUrl, body.toString(), {
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' }
      });

      PkceStore.delete(state);

      const { access_token, id_token, token_type, expires_in } = tokenRes.data ?? {};

      if (!access_token) {
        throw new InternalServerErrorException('Token response sem access_token');
      }

      // Validação do emissor via JWK do provedor
      const jwkUrl = `${baseUrl}/jwk`;
      let jwkKeys: Array<{ kid?: string; kty?: string; alg?: string; n?: string; e?: string }> = [];
      try {
        const jwkRes = await axios.get(jwkUrl);
        const body = jwkRes.data ?? {};
        jwkKeys = Array.isArray(body.keys) ? body.keys : [];
      } catch (e) {
        console.error(e);
        throw new BadRequestException('Falha ao consultar JWK do provedor');
      }

      const accessHeader = decodeJwtHeader(access_token) ?? {};
      const idHeader = id_token ? (decodeJwtHeader(id_token) ?? {}) : undefined;
      const accessKid = accessHeader.kid as string | undefined;
      const idKid = idHeader && (idHeader.kid as string | undefined);

      const kidExists = (kid?: string) => !!kid && jwkKeys.some(k => k.kid === kid);
      if (!kidExists(accessKid) || (idHeader && !kidExists(idKid))) {
        throw new BadRequestException('Falha na validação do token com JWK do provedor');
      }

      const claims = decodeJwt(access_token) ?? {};
      const expectedIss = `${baseUrl}/`;
      if (typeof claims.iss === 'string' && !claims.iss.startsWith(expectedIss)) {
        throw new BadRequestException('Token emitido por provedor diferente do configurado');
      }
      const exp = typeof claims.exp === 'number' ? claims.exp : undefined;

      const userInfoUrl = `${baseUrl}/userinfo`;
      const userRes = await axios.get(userInfoUrl, {
        headers: { Authorization: `Bearer ${access_token}` }
      });
      const raw = userRes.data ?? {};
      const user: GovbrUserInfo = {
        sub: raw.sub,
        name: raw.name ?? null,
        email: raw.email ?? null,
        email_verified: !!raw.email_verified,
        picture: raw.picture ?? null,
        profile: raw.profile ?? null,
        social_name: raw.social_name ?? null
      };
      const sessionId = randomBytes(18).toString('hex');
      const now = Date.now();
      const expires_at = exp ? exp * 1000 : expires_in ? now + expires_in * 1000 : undefined;

      const dataSource = getDataSource();
      const userRepo = dataSource.getRepository(Usuario);
      const cpf = typeof user.sub === 'string' ? user.sub.replace(/\D/g, '') : '';
      let usuario = await userRepo.findOne({ where: { nrCpf: cpf } });
      if (!usuario) {
        usuario = userRepo.create({ nrCpf: cpf, nmUsuario: user.name ?? '' });
      } else {
        usuario.nmUsuario = user.name ?? usuario.nmUsuario;
      }

      // Verifica se não existe nenhum admin no sistema
      const adminCount = await userRepo.count({ where: { cdRole: UsuarioRole.ADMIN } });
      if (adminCount === 0) {
        console.log(`Nenhum admin encontrado. Promovendo primeiro usuário (CPF: ${cpf}) a ADMIN.`);
        usuario.cdRole = UsuarioRole.ADMIN;
        usuario.stAprovado = true;
        usuario.dhAprovacao = new Date();
      }

      await userRepo.save(usuario);

      if (!usuario.stAprovado) {
        return {
          access_token,
          token_type: token_type ?? 'Bearer',
          expires_in: expires_in ?? (exp ? Math.max(exp * 1000 - now, 0) / 1000 : undefined),
          expires_at, // ms
          user,
          pendente: true,
          aprovado: false,
          role: usuario.cdRole
        };
      }

      await AuthSessionStore.set(sessionId, usuario.id, expires_at);

      return {
        access_token,
        token_type: token_type ?? 'Bearer',
        expires_in: expires_in ?? (exp ? Math.max(exp * 1000 - now, 0) / 1000 : undefined),
        expires_at, // ms
        user,
        session_id: sessionId,
        pendente: false,
        aprovado: true,
        role: usuario.cdRole
      };
    } catch (err: any) {
      console.error(err);
      const data = err?.response?.data;
      throw new BadRequestException(data ?? 'Falha na troca de token');
    }
  }

  async getUserInfo(accessToken: string): Promise<any> {
    const baseUrl = this.env.get('GOVBR_URL');
    const userInfoUrl = `${baseUrl}/userinfo`;
    const res = await axios.get(userInfoUrl, {
      headers: { Authorization: `Bearer ${accessToken}` }
    });
    return res.data;
  }

  async logout(sessionId: string): Promise<void> {
    if (!sessionId) return;
    await AuthSessionStore.delete(sessionId);
  }
}
